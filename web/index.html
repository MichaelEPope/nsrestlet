NSRestlet
Making connecting to Netsuite Restlets easy.
Step 1: Import NSRestlet

  var nsrestlet = require('nsrestlet');
Step 2: Use OAuth or NLAuth

  var accountSettings = {
    accountId: "PUT YOUR ACCOUNT ID HERE",
    tokenKey: "PUT YOUR TOKEN KEY HERE",
    tokenSecret: "PUT YOUR TOKEN SECRET HERE",
    consumerKey: "PUT YOUR CONSUMER KEY HERE",
    consumerSecret: "PUT YOUR CONSUMER SECRET HERE"
  };
                

  var accountSettings = {
    accountId: "PUT YOUR ACCOUNT ID HERE",
    email: "PUT YOUR EMAIL HERE",
    password: "PUT YOUR PASSWORD HERE",
    role: "PUT YOUR ROLE KEY HERE"
  };
Step 3: Set a URL or Script & Deployment IDs

  var urlSettings = {
    url: "https://12345.restlets.api.netsuite.com/app/site/hosting/restlet.nl?script=1&deploy=1"
  };
                

  var urlSettings = {
    script: "customscript_test_rlet",
    deployment: "customdeploy_test_rlet"
  }
                

  var urlSettings = {
    script: 142,
    deployment: 1
  }
Step 4: Recieve Data with Callbacks or Promises

  var myInvoices = nsrestlet.createLink(accountSettings, urlSettings);

  myInvoices.get({id: '12345'}, function(error, body)
  {
      console.log(body);
  });




                

  var myInvoices = nsrestlet.createLink(accountSettings, urlSettings);

  myInvoices.get({id: '12345'}).then(function(body)
  {
      console.log(body);
  })
  .catch(function(error)
  {
      console.log(error);
  });
Wanna give it a try? 

  npm install --save nsrestlet
